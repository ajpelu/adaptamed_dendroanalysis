1.1 Get Satellite data

Using Google Earth Engine, we extracted spectral information for all the pixels covering the field plots. Several products were obtained:

  • Temporal series of Vegetation Indexes NDVI and EVI from MOD13Q1. We used the following GEE code to get the EVI and NDVI data for each field plot. We obtained a total of 503 values for each field plot (2000-2021, 23 images per year, except 2000 year (only 20 images)). The data were saved as dendroadaptamed_iv_mod13q1.csv (see data/dendroadaptamed_iv_mod13q1.csv). The temporal resolution was one image each 16 days (23 images per year), and the spatial resoultion was 250 m.
var parcelas = ee.FeatureCollection("users/ajpelu/geoinfo_adaptamed_dendro");

// Define study period 
var startdate = ee.Date('2000-01-01');
var enddate = ee.Date('2021-12-31'); 

//-------------------- Load MODIS MOD13Q1 
// filter by date 
// select bands: EVI, NDVI, DOY, SummaryQA
var mod13 = ee.ImageCollection('MODIS/006/MOD13Q1')
  .filterDate(startdate, enddate)
  .select(['EVI', 'NDVI','DayOfYear']); 

print(mod13);


// Apply a function to every image of the collection (use .map()) 
// Get value of each band b of each temporal image t of MODQ13Q1
// for each of the coordinate provide in parcelas
var mydata = mod13.map(function(i) {
  return i.reduceRegions(parcelas, 'first');
});

// TODO: Complete doc 
var extract = ee.FeatureCollection(mydata.flatten());

// Export table
Export.table.toDrive({
  collection: extract,
  description: 'my_extract_data',
  fileFormat: 'CSV',
});
  • Temporal series of NPP from MOD17A3. We used the following GEE code to get the annual NPP data for each field plot. We obtained a total of 22 values for each field plot (2000-2021; 1 value per year). The data were saved as dendroadaptamed_npp_mod17a3_yearly.csv (see data/dendroadaptamed_npp_mod17a3_yearly.csv). Temporal resolution was yearly; Spatial resolution was 500 m. See here for a description of the method to obtain the NPP.
var parcelas = ee.FeatureCollection("users/ajpelu/geoinfo_adaptamed_dendro");

// Define study period 
var startdate = ee.Date('2000-01-01');
var enddate = ee.Date('2021-12-31'); 

//-------------------- Load MODIS MOD17A3HGF.006 
// filter by date 
// select bands: Npp, Npp_QC
var modNpp = ee.ImageCollection('MODIS/006/MOD17A3HGF')
  .filterDate(startdate, enddate)
  .select(['Npp', 'Npp_QC']); 

print(modNpp);


// Apply a function to every image of the collection (use .map()) 
// Get value of each band b of each temporal image t of MODIS
// for each of the coordinate provide in parcelas
var mydata = modNpp.map(function(i) {
  return i.reduceRegions(parcelas, 'first');
});

// TODO: Complete doc 
var extract = ee.FeatureCollection(mydata.flatten());

// Export table
Export.table.toDrive({
  collection: extract,
  description: 'my_extract_data',
  fileFormat: 'CSV',
});
